5)
Build host-independant toolchain (compiler, assembler, linker, libraries,...)

Compile cross binutils

    --prefix=<INSTALL>/tools        \
    --with-sysroot=/mnt/lfs        \
    --with-lib-path=<INSTALL>/tools/lib  \
    --target=i686-rastakware-gnu-gcc       \
    --disable-nls              \
    --disable-werror \
    && make && make install; }
